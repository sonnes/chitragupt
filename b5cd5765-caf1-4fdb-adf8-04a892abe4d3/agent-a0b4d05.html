        return nil, fmt.Errorf("no messages found in session")
    }

    messages := groupAndMapMessages(entries)
    first := entries[0]
    last := entries[len(entries)-1]

    createdAt := parseTime(first.Timestamp)
    var updatedAt *time.Time

        updatedAt = &t
